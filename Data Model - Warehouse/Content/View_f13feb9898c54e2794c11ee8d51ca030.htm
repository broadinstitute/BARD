DDL Code  CREATE VIEW BBDS_STAGING.EXPERIMENT_LOAD_BATCH_V
(EXPERIMENT_ID, LOAD_ORDER, LOAD_COST, LOAD_REASON) AS
select   a."EXPERIMENT_ID",a."LOAD_ORDER",a."LOAD_COST",
         -- see p_pre_load.determine_loadability
         decode(b.to_load,
             0, 'ERROR! - Shouldn''t Load',
             1, '1 - New Result File',
             2, '2 - Reload - Data Error Cleared',
             3, '3 - Reload - Updated Result File',
             4, '4 - Reload - Context Item Updated',
             5, '5 - Reload - Assay ID Changed',
             'UNKNOWN!'
         ) load_reason
from     EXPERIMENT_LOAD_BATCH a, STG_EXPERIMENT_FILE b
where    a.experiment_id = b.experiment_id
