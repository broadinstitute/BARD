DDL Code  CREATE VIEW BBDS.PERSON_ND
(PERSON_REF, PERSON_NAME, PHONE, INSTITUTE, JOB_TITLE, EFFECTIVE_FROM, SOURCE_PERSON_ID, PERSON_SOURCE_SYSTEM, EMAIL_ADDRESS) AS
select  
          person_ref
         ,person_name
         ,phone
         ,institute
         ,job_title
         ,effective_from
         ,source_person_id
         ,person_source_system
         ,email_address
from     (
     select a.*,
            row_number() over(partition by source_person_id order by effective_from desc ) rseq$
     from    person a
)
where    rseq$ = 1
