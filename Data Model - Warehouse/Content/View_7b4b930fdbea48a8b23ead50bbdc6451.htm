DDL Code  CREATE VIEW BBDS_STAGING.EXPERIMENT_LOAD_COST_V
(EXPERIMENT_ID, LOAD_COST, FILE_SIZE, N_RESULT) AS
select a.experiment_id,
        decode(b.n_result, null, floor(a.file_size/10), b.n_result) load_cost,
        a.file_size,
        b.n_result
from   bbds.X_EXPERIMENT_FILE a, EXPERIMENT_LOAD b
where a.experiment_id = b.experiment_id (+)
