DDL Code  CREATE VIEW BBDS.COMPOUND_ASSAY_AGG_ROW_HDR AS
SELECT RE.COMPOUND_REF, RE.OUTCOME, RE.SUBSTANCE_CNT, RE.SUBSTANCE_IDS
FROM BBDS.AGGREGATE_TYPE AG, BBDS.RESULT_AGGREGATE RE
WHERE RE.AGGREGATE_TYPE_REF = AG.AGGREGATE_TYPE_REF
   AND AG.AGGREGATE_COLUMNS = 'compound'
