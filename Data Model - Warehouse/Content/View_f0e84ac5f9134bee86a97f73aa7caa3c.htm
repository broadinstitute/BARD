DDL Code  CREATE VIEW BBDS_STAGING.TEST_GRID_ETL_IDENTIFIER
(PROJECT_ID, ASSAY_ID, EXPERIMENT_ID, ELEMENT_ID, "Project-Count_PASS", "Project-Count_TOBJ", "Assay-Count_PASS", "Assay-Count_TOBJ", "Annotation-in-Assay_PASS", "Annotation-in-Assay_TOBJ", "Assay-Experiment-Links_PASS", "Assay-Experiment-Links_TOBJ", "Experiment-Count_PASS", "Experiment-Count_TOBJ", "Result-Without-Annotation_PASS", "Result-Without-Annotation_TOBJ") AS
SELECT "PROJECT_ID","ASSAY_ID","EXPERIMENT_ID","ELEMENT_ID","Project-Count_PASS","Project-Count_TOBJ","Assay-Count_PASS","Assay-Count_TOBJ","Annotation-in-Assay_PASS","Annotation-in-Assay_TOBJ","Assay-Experiment-Links_PASS","Assay-Experiment-Links_TOBJ","Experiment-Count_PASS","Experiment-Count_TOBJ","Result-Without-Annotation_PASS","Result-Without-Annotation_TOBJ"
FROM (SELECT project_id, assay_id, experiment_id, element_id, test_object_id, test_name, pass_fail FROM test_object
       WHERE test_object_id IN
           (SELECT Max(test_object_id)
           FROM test_object
           WHERE test_category = 'ETL'
             AND (experiment_id IS NOT NULL
                 OR assay_id IS NOT NULL
                 OR project_id IS NOT NULL
                 OR element_id IS NOT NULL)
           GROUP BY project_id, assay_id, experiment_id, element_id, test_name)
       )
pivot (Min(pass_fail) AS pass,
        Max(Decode(pass_fail, 'Pass', '', test_object_id)) AS tobj
        FOR (test_name) IN ('Project-Count' AS "Project-Count",
                           'Assay-Count' AS "Assay-Count",
                           'Annotation-in-Assay' AS "Annotation-in-Assay",
                           'Assay-Experiment-Links' AS "Assay-Experiment-Links",
                           'Experiment-Count' AS "Experiment-Count",
                           'Result-Without-Annotation' AS "Result-Without-Annotation"
                           ))
ORDER BY project_id nulls last, assay_id nulls last, experiment_id, element_id
