DDL Code  CREATE VIEW BBDS.RESULT_TYPE_ND
(RESULT_TYPE_REF, RESULT_TYPE_ID, STATS_MODIFIER_ID, RESULT_TYPE_STATUS, RESULT_TYPE, STATS_MODIFIER, DISPLAY_PRIORITY, EFFECTIVE_FROM) AS
select
          result_type_ref
         ,result_type_id
         ,stats_modifier_id
         ,result_type_status
         ,result_type
         ,stats_modifier
         ,display_priority
         ,effective_from
from     (
     select a.*,
            row_number() over(partition by result_type_id, stats_modifier_id, display_priority order by effective_from desc) rseq$
     from    result_type a
)
where    rseq$ = 1
