DDL Code  CREATE VIEW BBDS_STAGING.TEST_GRID_ETL_EXPERIMENT
(PROJECT_ID, ASSAY_ID, EXPERIMENT_ID, ELEMENT_ID, "Assay-Experiment-Links_PASS", "Experiment-Count_PASS", "Result-Without-Annotation_PASS", "Annotation-in-Assay_PASS", "Assay-Count_PASS", "Project-Count_PASS") AS
SELECT "PROJECT_ID","ASSAY_ID","EXPERIMENT_ID","ELEMENT_ID","Assay-Experiment-Links_PASS","Experiment-Count_PASS","Result-Without-Annotation_PASS","Annotation-in-Assay_PASS","Assay-Count_PASS","Project-Count_PASS"
FROM (SELECT project_id, assay_id, experiment_id, element_id, test_name, pass_fail FROM test_object
       WHERE test_object_id IN
           (SELECT Max(test_object_id)
           FROM test_object
           WHERE test_category = 'ETL'
             AND (experiment_id IS NOT NULL
                 OR assay_id IS NOT NULL
                 OR project_id IS NOT NULL
                 OR element_id IS NOT NULL)
           GROUP BY project_id, assay_id, experiment_id, element_id, test_name)
       )
pivot (Min(Decode(pass_fail, 'Warning', 'Fail-warning', pass_fail)) AS pass
         FOR (test_name) IN ('Assay-Experiment-Links' AS "Assay-Experiment-Links",
                           'Experiment-Count' AS "Experiment-Count",
                           'Result-Without-Annotation' AS "Result-Without-Annotation",
                           'Annotation-in-Assay' AS "Annotation-in-Assay",
                           'Assay-Count' AS "Assay-Count",
                           'Project-Count' AS "Project-Count"))
ORDER BY project_id, assay_id, experiment_id, element_id
