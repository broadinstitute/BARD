DDL Code  CREATE VIEW BBDS_STAGING.TEST_GRID_ETL_DIMENSION
(DIMENSION, "Annotation-Set_PASS", "Annotation-Set_TOBJ", "Referential-Integrity_PASS", "Referential-Integrity_TOBJ", "Referential-Integrity-Batch_PA", "Referential-Integrity-Batch_TO") AS
SELECT "DIMENSION","Annotation-Set_PASS","Annotation-Set_TOBJ","Referential-Integrity_PASS","Referential-Integrity_TOBJ","Referential-Integrity-Batch_PA","Referential-Integrity-Batch_TO"
FROM (SELECT test_name, dimension, test_object_id, pass_fail FROM test_object
       WHERE test_object_id IN
           (SELECT Max(test_object_id)
           FROM test_object
           WHERE test_category = 'ETL'
             AND dimension IS NOT null
           GROUP BY dimension, test_name)
       )
pivot (Min(pass_fail) AS pass,
        LISTAGG(Decode(pass_fail, 'Pass', '', test_object_id), ', ') WITHIN GROUP (ORDER BY TEST_OBJECT_ID) AS tobj
        FOR (test_name) IN ('Annotation-Set' AS   "Annotation-Set",
                             'Referential-Integrity' AS "Referential-Integrity",
                             'Referential-Integrity-Batch' AS "Referential-Integrity-Batch"))
ORDER BY dimension
