DDL Code  CREATE VIEW BBDS_STAGING.V_EXPERIMENT_LOAD_ATTEMPT
(EXPERIMENT_ID, LOAD_IDS) AS
SELECT experiment_id,
     listagg(load_id, ', ') within GROUP (ORDER BY load_id DESC) load_ids
FROM load_detail
WHERE load_id IN
     (SELECT load_id FROM load_summary
     WHERE start_time > To_Date('01/01/2014', 'MM/DD/YYYY'))
   --AND result_STATUS != 'PENDING'
GROUP BY experiment_id
