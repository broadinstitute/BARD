DDL Code  CREATE OR REPLACE VIEW experiment_load_detail_v (
   experiment_id,
   annotation_status,
   n_annotation_set,
   n_annotation_context,
   t_annotation_load,
   result_status,
   n_result,
   n_result_loaded,
   t_result_load,
   t_result_delete,
   load_id,
   load_date
) AS
with latest_detail as (
     select   *
     from     (
       select a.*,
              row_number() over(partition by experiment_id order by load_id desc) rseq$
       from    load_detail a
     )
     where    rseq$ = 1
)
select
          a.experiment_id
         ,a.annotation_status
         ,a.n_annotation_set
         ,a.n_annotation_context
         ,a.t_annotation_load
         ,a.result_status
         ,a.n_result
         ,a.n_result_loaded
         ,a.t_result_load
         ,a.t_result_delete
         ,a.load_id
         ,b.end_time as load_date
from     latest_detail a, load_summary b
where    a.load_id = b.load_id
