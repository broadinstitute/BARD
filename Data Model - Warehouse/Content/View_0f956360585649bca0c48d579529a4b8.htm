DDL Code  CREATE VIEW BBDS_STAGING.V_LOG4O_TAIL
(ID, LOGGER, SESSION_ID, LOG_LEVEL, LOG_TIME, MESSAGE) AS
select   "ID","LOGGER","SESSION_ID","LOG_LEVEL","LOG_TIME","MESSAGE"
from (
   select * from log4o_log order by id desc
)
where    rownum <= 100
order by rownum desc
