DDL Code  CREATE VIEW BBDS_STAGING.LAST_LOAD_EXPT_RESULT_TYPE
(LOAD_ID, EXPERIMENT_ID, RESULT_TYPE_ID, STATS_MODIFIER_ID, DISPLAY_PRIORITY, RESULT_TYPE_FULLNAME, ANNOTATION_STATUS, RESULT_STATUS, LOAD_START) AS
select   load_id,
       experiment_id,
       result_type_id,
       stats_modifier_id,
       display_priority,
       result_type_fullname,
       annotation_status,
       result_status,
       load_start
     from     (
       select rt.result_type_id,
              rt.stats_modifier_id,
              rt.display_priority,
              rt.result_type_fullname,
              ls.start_time load_start,
              ld.*,
              row_number() over(partition by ld.experiment_id, rt.result_type_id, rt.stats_modifier_id, rt.display_priority order by ld.load_id desc) rseq$
       from    load_detail ld,
             load_summary ls,
           bbds.experiment e,
           bbds.experiment_result_type ert,
           bbds.result_type rt
       where ls.load_id = ld.load_id
         AND e.experiment_id = ld.experiment_id
         AND ert.experiment_ref = e.experiment_ref
         AND rt.result_type_ref = ert.result_type_ref) ld
     where rseq$ = 1
