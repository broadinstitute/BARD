DDL Code  CREATE VIEW BBDS_STAGING.SRC_ASSAY_ANNOTATION_V
(CONTEXT_ID, ASSAY_ID, EXPERIMENT_ID, CONTEXT_NAME, CONTEXT_GROUP, ATTRIBUTE_ID, ATTRIBUTE_NAME, VALUE_ID, EXT_VALUE_ID, VALUE_DISPLAY, VALUE_NUM, VALUE_MIN, VALUE_MAX, QUALIFIER, ANNOTATION_UNIT, ANNOTATION_UNIT_ID, ANNOTATION_LEVEL) AS
select   distinct
         b.assay_context_id context_id,
         b.assay_id,
         a.experiment_id,
         b.context_name,
         b.context_group,
         c.attribute_id,
         e.label attribute_name,
         c.value_id,
         c.ext_value_id,
         nvl(c.value_display, nvl(to_char(value_num), nvl(ext_value_id, to_char(value_id)))) value_display,
         c.value_num,
         c.value_min,
         c.value_max,
         nvl(c.qualifier, '=') qualifier,
         nvl(u.abbreviation, u.label) annotation_unit,
         e.unit_id annotation_unit_id,
         'ASSAY' annotation_level
from     experiment@BARD_CAP a,
         assay_context@BARD_CAP b,
         assay_context_item@BARD_CAP c,
         element@BARD_CAP e,
         element@BARD_CAP u
where    a.assay_id = b.assay_id
and      b.assay_context_id = c.assay_context_id
and      c.attribute_type = 'Fixed'
and      c.attribute_id = e.element_id
and      e.unit_id = u.element_id (+)
