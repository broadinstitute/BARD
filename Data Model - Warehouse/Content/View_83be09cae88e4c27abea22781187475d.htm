DDL Code  CREATE VIEW BBDS_STAGING.LAST_LOAD_EXPERIMENT
(LOAD_ID, EXPERIMENT_ID, ANNOTATION_STATUS, RESULT_STATUS, LOAD_START) AS
select   load_id,
       experiment_id,
       annotation_status,
       result_status,
       load_start
     from     (
       select ls.start_time load_start,
              ld.*,
              row_number() over(partition by experiment_id order by ld.load_id desc) rseq$
       from    load_detail ld,
              load_summary ls
       WHERE ls.load_id = ld.load_id
     )
     where    rseq$ = 1
