DDL Code  CREATE VIEW BBDS_STAGING.V_LOG4O_LAST_LOAD
(ID, LOGGER, SESSION_ID, LOG_LEVEL, LOG_TIME, MESSAGE) AS
SELECT "ID","LOGGER","SESSION_ID","LOG_LEVEL","LOG_TIME","MESSAGE"
FROM LOG4O_LOG
--WHERE log_level = 'ERROR'
WHERE id >= (SELECT Max(id)
     FROM LOG4O_LOG
     WHERE message = 'etl started')
ORDER BY id DESC
