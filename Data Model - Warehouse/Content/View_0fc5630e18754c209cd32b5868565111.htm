DDL Code  CREATE VIEW BBDS_STAGING.TEST_GRID_SOURCE
(PROJECT_ID, ASSAY_ID, EXPERIMENT_ID, ELEMENT_ID, "Annotation-in-Assay_PASS", "Annotation-in-Assay_TOBJ", "Element-Usage-Attribute_PASS", "Element-Usage-Attribute_TOBJ", "Element-Usage-Numeric_PASS", "Element-Usage-Numeric_TOBJ", "Element-Usage-Result_PASS", "Element-Usage-Result_TOBJ", "Element-attribute-value-desc_P", "Element-attribute-value-desc_T", "Result-Type_PASS", "Result-Type_TOBJ", "Result-Without-Annotation_PASS", "Result-Without-Annotation_TOBJ", "Result-file-counts-batch_PASS", "Result-file-counts-batch_TOBJ") AS
SELECT "PROJECT_ID","ASSAY_ID","EXPERIMENT_ID","ELEMENT_ID","Annotation-in-Assay_PASS","Annotation-in-Assay_TOBJ","Element-Usage-Attribute_PASS","Element-Usage-Attribute_TOBJ","Element-Usage-Numeric_PASS","Element-Usage-Numeric_TOBJ","Element-Usage-Result_PASS","Element-Usage-Result_TOBJ","Element-attribute-value-desc_P","Element-attribute-value-desc_T","Result-Type_PASS","Result-Type_TOBJ","Result-Without-Annotation_PASS","Result-Without-Annotation_TOBJ","Result-file-counts-batch_PASS","Result-file-counts-batch_TOBJ"
FROM (SELECT test_name, project_id, assay_id, experiment_id, element_id, test_object_id, pass_fail FROM test_object
       WHERE test_object_id IN
           (SELECT Max(test_object_id)
           FROM test_object
           WHERE test_category = 'Source'
           GROUP BY project_id, assay_id, experiment_id, element_id, test_name)
       )
pivot (Min(pass_fail) AS pass,
        Max(Decode(pass_fail, 'Pass', '', test_object_id)) AS tobj
        FOR (test_name) IN ('Annotation-in-Assay' AS "Annotation-in-Assay",
                             'Element-Usage-Attribute' AS "Element-Usage-Attribute",
                             'Element-Usage-Numeric' AS "Element-Usage-Numeric",
                             'Element-Usage-Result' AS "Element-Usage-Result",
                             'Element-attribute-value-desc' AS "Element-attribute-value-desc",
                             'Result-Type' AS "Result-Type",
                             'Result-Without-Annotation' AS "Result-Without-Annotation",
                             'Result-file-counts-batch' AS "Result-file-counts-batch"))
ORDER BY project_id nulls last, assay_id nulls last, experiment_id, element_id
