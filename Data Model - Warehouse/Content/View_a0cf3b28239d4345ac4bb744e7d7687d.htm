DDL Code  CREATE VIEW BBDS.COMPOUND_ASSAY_AGG_CELL AS
SELECT RE.Compound_ref,
      RE.Assay_ref,
      RE.result_type_ref,
      RE.VALUE_NUM,
      RE.VALUE_DISPLAY,
      RE.TOOLTIP_DISPLAY,
      RE.LOAD_DATE,
      RE.EXPERIMENT_CNT,
      RE.EXPERIMENT_IDS,
      RE.OUTCOME,
      RE.SUBSTANCE_CNT,
      RE.SUBSTANCE_IDS
FROM BBDS.AGGREGATE_TYPE AG, BBDS.RESULT_AGGREGATE RE
WHERE RE.Aggregate_Type_ref = AG.Aggregate_Type_ref
   AND AG.Aggregate_columns = 'assay, compound, result_type'
