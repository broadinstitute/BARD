PostSQL  -- create a set of 3 digit values

DECLARE
      ln_digit1       binary_integer;
      ln_digit2       binary_integer;
      ln_digit3       binary_integer;
      ln_digit4       binary_integer;
      ln_exp          binary_integer;
      ln_value_ref    NUMBER :=1;
      lv_neg          CHAR(1) := '-';
      lv_number       VARCHAR2(100);
      ln_duplicates   NUMBER := 1;

BEGIN
     -- start with the negative numbers
     FOR ln_exp IN REVERSE -15..14
     LOOP
         FOR ln_digit3 IN REVERSE 1..9
         LOOP
             FOR ln_digit2 IN REVERSE 0..9
             LOOP
                 FOR ln_digit1 IN REVERSE 0..9
                 LOOP
                     lv_number := lv_neg || ln_digit3 || '.' || ln_digit2 || ln_digit1 || 'E' || ln_exp;
                     --Dbms_Output.put_line (lv_number || ';   '|| To_Char(To_Number(lv_number)));
                     INSERT INTO Value
                         (VALUE_REF,
                          VALUE_NUM,
                           Exponent,
                           VALUE_TEXT)
                     values
                         (ln_value_ref,
                         To_Number(lv_number),
                         ln_exp,
                         lv_number);

                     ln_value_ref := ln_value_ref + 1;
                 END LOOP;
             END LOOP;
         END LOOP;
     END LOOP;

     -- insert a zero at the middle
     INSERT INTO Value
       (VALUE_REF,
       VALUE_NUM,
       Exponent,
       VALUE_TEXT)
     VALUES
       (ln_value_ref,
        0,
        0,
        '0');
     ln_value_ref := ln_value_ref + 1;

     -- and now the positive numbers
     FOR ln_exp IN -15..14
     LOOP
         FOR ln_digit3 IN 1..9
         LOOP
             FOR ln_digit2 IN 0..9
             LOOP
                 FOR ln_digit1 IN 0..9
                 LOOP
                     lv_number := ln_digit3 || '.' || ln_digit2 || ln_digit1 || 'E' || ln_exp;
                     INSERT INTO Value
                         (VALUE_REF,
                          VALUE_NUM,
                           Exponent,
                           VALUE_TEXT)
                     values
                         (ln_value_ref,
                         To_Number(lv_number),
                         ln_exp,
                         lv_number);

                     ln_value_ref := ln_value_ref + 1;
                  END LOOP;
             END LOOP;
         END LOOP;
     END LOOP;

     -- remove any duplicates that have appeared from the rounding
     LOOP
         delete FROM Value
         WHERE value_ref IN (
             SELECT Max(value_ref) FROM Value
             GROUP BY value_num
             HAVING Count(*) >1);

         EXIT WHEN SQL%ROWCOUNT = 0;
      END LOOP;

      -- and clean up the value_ref index numbers
      -- to put the table into numerical order
      UPDATE Value v1
      SET value_ref = (SELECT Count(*)
               FROM Value v2
               WHERE v2.value_num <= v1.value_num);

      -- clean up the text representation of the rounding errors
      UPDATE Value
      SET value_text = To_Char(value_num, '0.99EEEE');

     commit;

END;
