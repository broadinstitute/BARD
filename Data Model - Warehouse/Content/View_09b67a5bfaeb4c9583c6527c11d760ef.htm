DDL Code  CREATE VIEW BBDS_STAGING.EXPERIMENT_LOAD_DETAIL_V
(EXPERIMENT_ID,
      ANNOTATION_STATUS,
      N_ANNOTATION_SET,
      N_ANNOTATION_CONTEXT,
      T_ANNOTATION_LOAD,
      RESULT_STATUS,
      N_RESULT,
      N_RESULT_LOADED,
      T_RESULT_LOAD,
      T_RESULT_DELETE,
      LOAD_ID,
      LOAD_DATE) AS
with latest_detail as (
     select   *
     from (select a.*,
              row_number() over(partition by experiment_id order by load_id desc) rseq$
              from    load_detail a)
     where    rseq$ = 1)
select
          a.experiment_id
         ,a.annotation_status
         ,a.n_annotation_set
         ,a.n_annotation_context
         ,a.t_annotation_load
         ,a.result_status
         ,a.n_result
         ,a.n_result_loaded
         ,a.t_result_load
         ,a.t_result_delete
         ,a.load_id
         ,b.end_time as load_date
from     latest_detail a, load_summary b
where    a.load_id = b.load_id
