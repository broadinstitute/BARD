DDL Code  CREATE VIEW BBDS.COMPOUND_ASSAY_AGG
(Compound_ref, Assay_ref, result_type_ref, VALUE_NUM, VALUE_DISPLAY, TOOLTIP_DISPLAY, LOAD_DATE, EXPERIMENT_CNT, EXPERIMENT_IDS, OUTCOME_REF, SUBSTANCE_CNT, SUBSTANCE_IDS, COL_EXPERIMENT_CNT, COL_EXPERIMENT_IDS, ROW_SUBSTANCE_CNT, ROW_SUBSTANCE_IDS) AS
SELECT CO.COMPOUND_REF, CO.ASSAY_REF, CO.RESULT_TYPE_REF, CO.VALUE_NUM, CO.VALUE_DISPLAY, CO.TOOLTIP_DISPLAY, CO.LOAD_DATE, CO.EXPERIMENT_CNT, CO.EXPERIMENT_IDS, CO.OUTCOME_REF, CO.SUBSTANCE_CNT, CO.SUBSTANCE_IDS, CL_HDR.EXPERIMENT_CNT, CL_HDR.EXPERIMENT_IDS, RW_HDR.SUBSTANCE_CNT, RW_HDR.SUBSTANCE_IDS
FROM COMPOUND_ASSAY_AGG_CELL CO
left outer join COMPOUND_ASSAY_AGG_COL_HDR CL_HDR
   ON(CO.ASSAY_REF = CL_HDR.ASSAY_REF
   AND CO.RESULT_TYPE_REF = CL_HDR.RESULT_TYPE_REF)
left outer join COMPOUND_ASSAY_AGG_ROW_HDR RW_HDR
   on (CO.COMPOUND_REF = RW_HDR.COMPOUND_REF)
