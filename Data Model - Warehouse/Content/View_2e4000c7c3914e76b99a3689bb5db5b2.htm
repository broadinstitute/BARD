DDL Code  CREATE VIEW BBDS.X_ANNOTATION
(ANNOTATION_REF, CONTEXT_ID, CONTEXT_NAME, CONTEXT_GROUP, ATTRIBUTE_ID, ATTRIBUTE_NAME, VALUE_ID, EXT_VALUE_ID, VALUE_DISPLAY, VALUE_NUM, VALUE_MIN, VALUE_MAX, QUALIFIER, ANNOTATION_UNIT, ANNOTATION_UNIT_ID, CONTEXT_REF, ANNOTATION_SET_REF, ANNOTATION_LEVEL) AS
SELECT
   sac.annotation_ref,
   sac.ASSAY_CONTEXT_ID,
   DECODE(Upper(sac.CONTEXT_NAME), '', NULL, sac.context_name),
   DECODE(Upper(sac.CONTEXT_group), '', NULL, sac.context_group),
   sac.ATTRIBUTE_ID,
   sac.ATTRIBUTE_NAME,

   DECODE(sac.VALUE_ID, -100, NULL, sac.value_id),
   DECODE(Upper(sac.EXT_VALUE_ID), '', NULL, sac.EXT_VALUE_ID),
   DECODE(Upper(sac.VALUE_DISPLAY), '', NULL, sac.VALUE_DISPLAY),
   sac.VALUE_NUM,
   sac.VALUE_MIN,
   sac.VALUE_MAX,
   sac.QUALIFIER,
   DECODE(Upper(sac.ANNOTATION_UNIT), '', NULL, sac.ANNOTATION_UNIT),
   DECODE(sac.ANNOTATION_UNIT_ID, -200, NULL, sac.ANNOTATION_UNIT_ID),
   ST.CONTEXT_REF,
   ST.ANNOTATION_SET_REF,
   SAC.annotation_level
FROM
   X_ANNOTATION_CONTEXT_LKP sac,
   X_ANNOTATION_SET_CONTEXT_LKP ST
WHERE
   ST.CONTEXT_REF = SAC.CONTEXT_REF
