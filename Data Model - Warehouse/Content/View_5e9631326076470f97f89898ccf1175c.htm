DDL Code  CREATE VIEW BBDS_STAGING.LOAD_SUMMARY_V
(LOAD_ID, START_TIME, RUN_TIME, EXPERIMENT_COUNT, N_RESULT, N_RESULT_LOADED, ANNOTATION_LOAD_TIME, RESULT_LOAD_TIME, RESULT_DELETE_TIME, ANNOTATION_ERRORS, RESULT_ERRORS, EXPERIMENT_ERRORS, LOG_ERRORS, LOG_WARNS) AS
select    a.load_id
--         ,a.session_id
         ,a.start_time
         ,p_log4o.format_elapsed_time((a.end_time-a.start_time)*24*3600) run_time
         ,b.experiment_count
         ,b.n_result
         ,b.n_result_loaded
         ,p_log4o.format_elapsed_time(a.annotation_load_time) annotation_load_time
         ,p_log4o.format_elapsed_time(a.result_load_time)      result_load_time
         ,p_log4o.format_elapsed_time(b.result_delete_time)      result_delete_time
         ,b.annotation_errors
         ,b.result_errors
         ,c.experiment_errors
         ,a.log_errors
         ,a.log_warns
from     load_summary a,
         (
             select   x.load_id
                     ,count(*)    experiment_count
                     ,sum(x.n_result) n_result
                     ,sum(x.n_result_loaded) n_result_loaded
                     ,sum(decode(x.annotation_status, 'ERROR', 1, 0)) annotation_errors
                     ,sum(decode(x.result_status, 'ERROR', 1, 0)) result_errors
                     ,sum(x.t_result_delete) result_delete_time
             from     load_detail x
             group by x.load_id
         ) b,
         (
             select   x.load_id,
                     sum(decode(e.submission_version, 555, 1, 666, 1, 0)) experiment_errors
             from     bbds.experiment e, load_detail x
             where    e.experiment_id = x.experiment_id
             group by x.load_id
         ) c
where    a.load_id = b.load_id
and      a.load_id = c.load_id (+)
