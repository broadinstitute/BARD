Definition 
 
DDL Code  CREATE VIEW BBDS.COMPOUND_ASSAY_AGG_ALL
(COMPOUND_REF, ASSAY_REF, RESULT_TYPE_REF, VALUE_NUM, VALUE_DISPLAY, TOOLTIP_DISPLAY, LOAD_DATE, EXPERIMENT_CNT, EXPERIMENT_IDS, OUTCOME, SUBSTANCE_CNT, SUBSTANCE_IDS) AS
SELECT caa.compound_ref,
     caa.assay_ref,
     result_type_ref,
     value_num,
     value_display,
     tooltip_display,
     load_date,
     ass.all_experiment_cnt,
     ass.all_experiment_ids,
     outcome,
     (select count(s.substance_id)
       from bbds.substance s,
           bbds.compound c
       where c.compound_id = s.compound_id
         AND c.compound_ref = caa.compound_ref) all_substance_cnt,
     (select listagg(substance_id, chr(10)) within group (order by substance_id)
       from bbds.substance s,
           bbds.compound c
       where c.compound_id = s.compound_id
         AND c.compound_ref = caa.compound_ref) all_substance_ids
FROM compound_assay_agg caa,
     (select assay_ref,
           count(e.experiment_id) all_experiment_cnt,
           listagg(experiment_id, chr(10)) within group (order by experiment_id) all_experiment_ids
       from bbds.assay a,
           bbds.experiment e
       where a.assay_id = e.assay_id
         and e.experiment_status = 'Approved'
         GROUP BY assay_ref) ass
WHERE ass.assay_ref (+) = caa.assay_ref
