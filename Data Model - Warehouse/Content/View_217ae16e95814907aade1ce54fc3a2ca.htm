DDL Code  CREATE VIEW BBDS.ASSAY_ND
(ASSAY_REF, ASSAY_ID, ASSAY_STATUS, ASSAY_SHORT_NAME, ASSAY_NAME, ASSAY_VERSION, ASSAY_OWNER, ACL_TICKET, EFFECTIVE_FROM, ASSAY_CLASS, SOURCE_SYSTEM, LOAD_DATE) AS
select  
      assay_ref
     ,assay_id
     ,assay_status
     ,assay_short_name
     ,assay_name
     ,assay_version
     ,assay_owner
     ,acl_ticket
     ,effective_from
     ,assay_class
     ,source_system
     ,load_date
from     (
     select a.*,
            -- add assay_ref as secondary sort in case of identical effective_from
            row_number() over(partition by assay_id order by effective_from desc, assay_ref desc) rseq$
     from    assay a
)
where    rseq$ = 1
