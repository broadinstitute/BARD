DDL Code  CREATE VIEW BBDS.PROJECT_ND
(PROJECT_REF, SOURCE_PROJECT_ID, PROJECT_NAME, GRANT_NUMBER, FUNDING_SOURCE, ACL_TICKET, EFFECTIVE_FROM, PROJECT_SOURCE_SYSTEM, INTENDED_MODE) AS
select
          project_ref
         ,source_project_id
         ,project_name
         ,grant_number
         ,funding_source
         ,acl_ticket
         ,effective_from
         ,project_source_system
         ,intended_mode
from     (
     select a.*,
            row_number() over(partition by source_project_id order by effective_from desc) rseq$
     from    project a
)
where    rseq$ = 1
