DDL Code  CREATE TABLE MLBD_DEV.Result(
     Result_ID             BIGINT           AUTO_INCREMENT,
     Value_Display         VARCHAR(256),
     value_num             FLOAT(8, 0),
     value_min             FLOAT(8, 0),
     Value_Max             FLOAT(8, 0),
     Qualifier             CHAR(2),
     Result_Status_ID      BIGINT           NOT NULL,
     Experiment_ID         BIGINT           NOT NULL,
     Substance_ID          BIGINT           NOT NULL,
     Result_Context_ID     BIGINT           NOT NULL,
     Entry_Unit            VARCHAR(100),
     Result_Type_ID        BIGINT           NOT NULL,
     Version               INT              DEFAULT 0 NOT NULL,
     Date_Created          TIMESTAMP        DEFAULT current_timestamp NOT NULL,
     Last_Updated          DATETIME,
     Modified_by           VARCHAR(40),
     PRIMARY KEY (Result_ID),
     CONSTRAINT FK_result_result_status FOREIGN KEY (Result_Status_ID)
     REFERENCES MLBD_DEV.Result_Status(Result_Status_ID),
     CONSTRAINT FK_result_experiment FOREIGN KEY (Experiment_ID)
     REFERENCES MLBD_DEV.Experiment(Experiment_ID),
     CONSTRAINT fk_result_result_context FOREIGN KEY (Result_Context_ID)
     REFERENCES MLBD_DEV.Result_Context(Result_Context_ID),
     CONSTRAINT FK_Result_substance FOREIGN KEY (Substance_ID)
     REFERENCES MLBD_DEV.Substance(Substance_ID),
     CONSTRAINT FK_result_unit FOREIGN KEY (Entry_Unit)
     REFERENCES MLBD_DEV.Unit(Unit),
     CONSTRAINT FK_Result_result_type FOREIGN KEY (Result_Type_ID)
     REFERENCES MLBD_DEV.Result_Type(Result_Type_ID),
     CONSTRAINT FK_Result_Qualifier FOREIGN KEY (Qualifier)
     REFERENCES MLBD_DEV.Qualifier(Qualifier)
)ENGINE=MYISAM
;
