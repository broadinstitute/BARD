DDL Code  CREATE TABLE MLBD.Experiment(
     Experiment_ID            INT               AUTO_INCREMENT,
     Experiment_Name          VARCHAR(256)      NOT NULL,
     Assay_ID                 INT               NOT NULL,
     Project_ID               INT,
     Experiment_Status_ID     INT               NOT NULL,
     Run_Date                 DATETIME,
     Hold_Until_Date          DATE,
     Description              VARCHAR(1000)     NOT NULL,
     Source_ID                INT               NOT NULL,
     CONSTRAINT CK_Hold_Until_date CHECK (Hold_Until_Date <= sysdate + 366),
     PRIMARY KEY (Experiment_ID),
     CONSTRAINT FK_experiment_assay FOREIGN KEY (Assay_ID)
     REFERENCES MLBD.Assay(Assay_ID),
     CONSTRAINT FK_Project_experiment FOREIGN KEY (Project_ID)
     REFERENCES MLBD.Project(Project_ID),
     CONSTRAINT FK_experiment_exprt_status FOREIGN KEY (Experiment_Status_ID)
     REFERENCES MLBD.Experiment_Status(Experiment_Status_ID),
     CONSTRAINT FK_Experiment_source_lab FOREIGN KEY (Source_ID)
     REFERENCES MLBD.Laboratory(Lab_ID)
)ENGINE=MYISAM
;
